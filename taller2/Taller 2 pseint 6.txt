Algoritmo TALLER_DOS_PUNTO_SEIS
	
	Definir tipoMoto Como Caracter;
	Definir obs_Moto_Alto_Cil Como Caracter;
	Definir obs_Moto_Bajo_Cil Como Caracter;
	Definir consMoto Como Caracter;
	Definir fecha_Ing_Alto Como Caracter;
	Definir fecha_Sal_Alto Como Caracter;
	Definir reparaciones_Alto Como Caracter;
	Definir fecha_Ing_Bajo Como Caracter;
	Definir fecha_Sal_Bajo Como Caracter;
	Definir reparaciones_Bajo Como Caracter;
	Definir opcion_Moto Como Caracter;
	

	Escribir "Digite ingreso si va a ingresar nueva moto a taller o Digite: consulta si va a consultar registros de motos en taller"
	leer opcion_Moto;
	
	si(opcion_Moto == "ingreso")=Verdadero Entonces
		
		Escribir "Digite: alto para ingresar moto de Alto Cilindraje, o Digite: bajo para ingresar moto de Bajo Cilindraje";
		leer tipoMoto;
		
		si(tipoMoto == "alto")= Verdadero Entonces
			Escribir "registre fecha de ingreso de la moto"
			leer fecha_Ing_Alto ;
			
			Escribir "Digite las observaciones de la moto"
			leer obs_Moto_Alto_Cil;
			
			Escribir "registre fecha de salida de la moto"
			leer fecha_Sal_Alto;
			
			Escribir "Digite las reparaciones de la moto"
			leer reparaciones_Alto;
		FinSi
		
		si(tipoMoto == "bajo")= Verdadero Entonces
			Escribir "registre fecha de ingreso de la moto"
			leer fecha_Ing_Bajo;
			
			Escribir "Digite las observaciones de la moto"
			leer obs_Moto_Bajo_Cil;
			
			Escribir "registre fecha de salida de la moto"
			leer fecha_Sal_Bajo;
			
			Escribir "Digite las reparaciones de la moto"
			leer reparaciones_Bajo;
		FinSi
		
		Escribir "Digite ingreso si va a ingresar nueva moto a taller o Digite: consulta si va a consultar registros de motos en taller"
		leer consMoto;
		
		
		si(consMoto == "consulta")=Verdadero Entonces
			
			Escribir "Digite: cons_bajo para consultar moto de Bajo Cilindraje, o Digite: cons_alto para ingresar moto de Alto Cilindraje";
			leer opcion_Moto;
			si(opcion_Moto == "cons_alto")= Verdadero Entonces
				Escribir "Su consulta para moto de Alto cilindraje es: fecha de ingreso: ",fecha_Ing_Alto, " / observaciones: ", obs_Moto_Alto_Cil, " / fecha de salida: ",fecha_Sal_Alto," / reparaciones: ",reparaciones_Alto;
			FinSi
			
			si(opcion_Moto == "cons_bajo")= Verdadero Entonces
				Escribir "Su consulta para moto de Bajo cilindraje es: fecha de ingreso: ",fecha_Ing_Bajo, " / observaciones: ", obs_Moto_Bajo_Cil, " / fecha de salida: ",fecha_Sal_Bajo," / reparaciones: ",reparaciones_Bajo;
			FinSi
			
		FinSi
		
		
		
		
	FinSi
	
	
	
	
	
	
FinAlgoritmo
