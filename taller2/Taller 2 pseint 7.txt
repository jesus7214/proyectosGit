Algoritmo TALLER_DOS_PUNTO_SIETE
	
	Definir peso_kg Como Real;
	Definir estatura_mt Como Real;
	Definir imc Como Real;
	
	
	Escribir "Ingrese su Peso en kg";
	Leer peso_kg;
	Escribir "ingrese su Estatura en metros" ;
	Leer estatura_mt;
	
	imc <- peso_kg / (estatura_mt * estatura_mt);
	
	Escribir "Su IMC es de: ", imc;
	
	si(imc < 18.5)=Verdadero Entonces
		Escribir "Su Peso es Bajo"
	FinSi
	
	si(imc > 18.5 y imc < 24.9)=Verdadero Entonces
		Escribir "Su Peso es Normal"
	FinSi
	
	si(imc > 25 y imc < 29.9)=Verdadero Entonces
		Escribir "Tiene Sobrepeso"
	FinSi
	
	si(imc >= 30)=Verdadero Entonces
		Escribir "Tiene Obesidad"
	FinSi
	
	
FinAlgoritmo
