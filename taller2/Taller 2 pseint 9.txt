Algoritmo TALLER_DOS_PUNTO_NUEVE
	
	Definir ladoA Como Entero;
	Definir ladoB Como Entero;
	
	Definir base Como Entero;
	Definir altura Como Entero;
	
	Definir baseMayor Como Entero;
	Definir baseMenor Como Entero;
	Definir altura_trap Como Entero;
	
	Definir op Como Caracter;
	
	Definir resultadoA_Tri Como Entero;
	Definir resultadoA_Rec Como Entero;
	Definir resultadoA_Trap Como Entero;

	
	Escribir "Para calcular Area de Triangulo Digite: triangulo // Para calcular Area de Reactangulo Digite: rectangulo // Para calcular Area de Trapecio Digite: trapecio";
	leer op;
	
	Si(op == "triangulo")=Verdadero Entonces
		Escribir "Va a calcular Área de Truangulo";
		Escribir "Digite el número de la base:"
		leer base;
		Escribir "Digite el número de la altura:"
		leer altura;
		resultadoA_Tri <- (base * altura)/2;
		Escribir "El área del Triángulo es: ", resultadoA_Tri;
	FinSi
	
	Si(op == "rectangulo")=Verdadero Entonces
		Escribir "Va a calcular Área de Rectangulo";
		Escribir "Digite el número del ladoA:"
		leer ladoA;
		Escribir "Digite el número del ladoB:"
		leer ladoB;
		resultadoA_Rec <- ladoA * ladoB;
		Escribir "El área del Rectangulo es: ", resultadoA_Rec;
	FinSi
	
	Si(op == "trapecio")=Verdadero Entonces
		Escribir "Va a calcular Área del Trapecio";
		Escribir "Digite el número dela baseMayor:"
		leer baseMayor;
		Escribir "Digite el número dela baseMenor:"
		leer baseMenor;
		Escribir "Digite el número de la altura:"
		leer altura_trap;
		resultadoA_Trap <- ((baseMenor * baseMayor) / 2 ) * altura_trap;
		Escribir "El área del Trapecio es: ", resultadoA_Trap
	FinSi
	
	
FinAlgoritmo
