Algoritmo TALLER_DOS_PUNTO_CUATRO
	
	Definir nombreUsuario Como Caracter;
	Definir respuesta Como Caracter;
	Definir respuestaDispo Como Caracter;
	Definir respuestaRecibir Como Caracter;
	Definir respuestaDaños Como Caracter;
	Definir nombrePeliculaDisp_A Como Caracter;
	Definir nombrePeliculaDisp_B Como Caracter;
	
	nombrePeliculaDisp_A <- "avatar";
	nombrePeliculaDisp_B <- "Matrix";
	
	Escribir "por favor, escriba su nombre";
	leer nombre;
	
	Escribir "Desea alquilar una pelicula?"
	leer respuesta ;
	Si (respuesta=="si")= Verdadero Entonces
		
		Escribir "Desea ver las peliculas disponibles?"
		leer respuestaDispo ;
		Si (respuestaDispo=="si")= Verdadero Entonces
			Escribir "Las películas disponibles en el momento son; ",nombrePeliculaDisp_A, " y ", nombrePeliculaDisp_B ;
		FinSi
		
		Escribir "Desea recibir película en la video tienda? ";
		leer respuestaRecibir ;
		Si(respuestaRecibir=="si")= Verdadero Entonces
			Escribir "recibirás la película en la video tienda";
		SiNo
			Escribir "la recibirás por otro medio distinto";
		FinSi
		
		Escribir "Desea realizar anotaciones sobre estas si se llegan a presentar daños u otra novedad sobre la película?";
		leer respuestaDaños ;
		si(respuestaDaños=="si")= Verdadero Entonces;
			Escribir "SI Podrás hacer comentarios obre el estado físico de las peliculas";
		SiNo
			Escribir "No podrás hacer comentarios sobre el estado de las películas";
		FinSi
		
	SiNo
		Escribir "No alquilaste nada, adios"
	FinSi
	
	
FinAlgoritmo
