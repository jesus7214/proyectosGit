	Definir placaE Como Caracter;

	Definir marcaC Como Caracter;
	Definir marcaD Como Caracter;

				Leer opcEscoger;
				Escribir " ------"

					si(espacioB == null)
						
						
						marcaB= marc();
						propietarioB= nom();
						el_PropietarioB= tel();
						placaB= plac();
						espacioB = placaB;

				FinSi
				
				Escribir " ------"

				Escribir " ------"

						propietarioC= nom();

						placaC= plac();
						espacioC = placaC;

				FinSi
				
				Escribir " ------"
				Escribir "Va a guardar Vehiculo? Digite: 1 para si // 2 para Reetirar // 3 para Consultar o // 4 para salir del menú"
				Leer opcEscoger;
				Escribir " ------"

						propietarioD= nom();
